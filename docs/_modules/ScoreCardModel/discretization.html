[docs]    def transform(self, x):
        """
        Parameters:

            x (Sequence): - 用于分段的序列

        Returns:

            np.array: - 以分段字符串为枚举内容的numpy序列

            
        """
        s = pd.cut(x, bins=self.bins)
        d = pd.get_dummies(s)
        z = d.T.to_dict()
        re = []
        for i, v in z.items():
            for j, u in v.items():
                if u == 1:
                    re.append(j)
        return np.array(re)
