        """计算分布


        Parameters:
            x (np.ndarray): - 复杂的多层序列
            segment (int): -  均分的分段数,默认100
        Returns:
            Tuple[np.ndarray, np.ndarray]: - 返回用于画图的X,Y
        """
        min_val = min(x)
        max_val = max(x)
        step = (min_val - max_val) / segment
        Y, X = np.histogram(x, bins=segment, range=(min_val, max_val))
        return Y, X
